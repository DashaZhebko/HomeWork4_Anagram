class Solution {
    public boolean isAnagram(String s, String t) {
        char [] masS = s.toCharArray();
        char [] masT = t.toCharArray();
        Arrays.sort(masS);
        Arrays.sort(masT);
        
        if (masS.length !=masT.length){
            return false;
        }
        return Arrays.equals(masT, masS);
        
    }
}
